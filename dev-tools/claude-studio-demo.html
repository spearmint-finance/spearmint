from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.orm import Session

from ..dependencies import get_db
from ..schemas.scenario import (
    ScenarioPreviewRequest,
    ScenarioPreviewResponse
)
from ...services.scenario_service import ScenarioService

router = APIRouter(
    prefix="/scenarios",
    tags=["scenarios"]
)


@router.post(
    "/preview",
    response_model=ScenarioPreviewResponse,
    status_code=status.HTTP_200_OK,
    summary="Preview financial scenario",
    description="Simulate a what-if scenario without saving it to the database"
)
def preview_scenario(
    request: ScenarioPreviewRequest,
    db: Session = Depends(get_db)
) -> ScenarioPreviewResponse:
    """
    Simulate a financial scenario and return baseline vs scenario comparison.

    This endpoint is deterministic and fast - it doesn't save anything to the database.
    Frontend can use this for real-time scenario exploration.

    Args:
        request: Scenario configuration with adjusters
        db: Database session

    Returns:
        Baseline and scenario projections with KPIs and deltas

    Raises:
        HTTPException: 400 if scenario configuration is invalid
        HTTPException: 500 if projection calculation fails
    """
    try:
        service = ScenarioService(db)
        result = service.preview_scenario(
            horizon_months=request.horizon_months,
            starting_balance=request.starting_balance,
            adjusters=request.adjusters
        )
        return result

    except ValueError as e:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail=str(e)
        )

    except Exception as e:
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail=f"Failed to generate scenario preview: {str(e)}"
        )
